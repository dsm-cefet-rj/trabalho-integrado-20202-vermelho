import React, { useEffect }  from 'react'

import Pedido from './Pedido.jsx'

import { useSelector, useDispatch } from 'react-redux'
import { updatePedidoServer, deletePedidoServer, addPedidoServer, fetchPedidos, selectAllPedidos } from '../store/reducers/pedidosrecebidosSlice'

import '../styles/PedidosRecebidos.css'

/**
 * @module components/PedidosRecebidos
 */

/**
 * Componente que instancia os pedidos com um componente filho e exibe na tela do app.
 */

function PedidosRecebidos() {
    const pedidos = useSelector(selectAllPedidos)
    const state = useSelector(state => state.pedidosrecebidos.status)
    const error = useSelector(state => state.pedidosrecebidos.error)

    const dispatch = useDispatch()

    useEffect(() => { // Faz o fetch quando pedidosrecebidos é renderizado
        if (state === "not_loaded" || state === "saved" || state === "deleted") {
            dispatch(fetchPedidos())
        } else if (state === 'failed') {
            setTimeout(() => dispatch(fetchPedidos()), 5000)
        }
    }, [state, dispatch])


    const Renderiza = pedido => {
        const lidaPronto = () => { // Destructuring do objeto pronto para atualizar com pronto alterado.
            const { id, numero, quarto, pedidos, total, pronto } = pedido
            dispatch(updatePedidoServer(
                {
                    id: id,
                    numero: numero,
                    quarto: quarto,
                    pedidos: pedidos,
                    total: total,
                    pronto: !pronto
                }
            )) 
        }

        if (!pedido.pronto) {
            return (
                <Pedido numero={pedido.numero} quarto={pedido.quarto} pedidos={pedido.pedidos}
                    total={pedido.total} pronto={pedido.pronto} lidaPronto={lidaPronto}></Pedido>
            )
        }
    }

    const Condicao = pedidos => {
        let renderiza = '';
        
        if (state === "loaded") {
            renderiza = <div>{pedidos.map(Renderiza)}</div>
        }else if (state === "loading") {
            renderiza = <div className = 'renderiza-aviso' id = 'loading'><br/><img id="loading-img" src="/images/cardapio2.png"/><div className= 'loading-div'> Carregando...</div> </div>
        }else if (state === "failed") {
            renderiza = <div className = 'renderiza-aviso' id = 'error'><div className='error-div'> Error: {error}</div> </div>
        }
       return renderiza
    }

    return (
        <div id="pedidos-recebidos">
            <div className="pedidos-recebidos-titulo">
                <h3>Pedidos Recebidos
                    <img className='refresh' src="/images/refresh.png" onClick = {() => dispatch(fetchPedidos())}/>
                </h3>
            </div>
            {state === undefined ? // Caso não pegue nada do backend
            <div>
              Sem novos pedidos!
            </div>
            :  
             <div>
                {Condicao(pedidos)}
            </div>}
          
            <br />
        </div>

    )
}

export default PedidosRecebidos;
